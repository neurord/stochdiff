173     public static String getShortSource() {
174         StackTraceElement[] stea = (new Exception()).getStackTrace();
175         String ss = (" at " + stea[2].toString());
176         if (ss.equals(lastShortSource)) {
177             ss = "";
178         } else {
179             lastShortSource = ss;
180         }
181         return ss;
182     }
